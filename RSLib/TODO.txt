public static Vector2 Rotate(this Vector2 vector, float degrees)
{
   float sin = Mathf.Sin(degrees * Mathf.Deg2Rad);
   float cos = Mathf.Cos(degrees * Mathf.Deg2Rad);

   float tx = vector.x;
   float ty = vector.y;
   vector.x = (cos * tx) - (sin * ty);
   vector.y = (sin * tx) + (cos * ty);
   return vector;
}
